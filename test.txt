

The issue is that the black screen browser is launching too early and overlapping with the video. Let's fix the timing in your startup script:

Replace with this corrected version:

#!/bin/bash
#
# Chatty AI Startup Script - Fixed Timing
#

# Configuration
VIDEO_PATH="/home/nickspi5/Chatty_AI/Chatty_AI_starting.mp4"
LOADING_PAGE="/home/nickspi5/Chatty_AI/templates/chatty_loading.html"
BLACK_PAGE="/home/nickspi5/Chatty_AI/templates/black.html"
CHATTY_URL="http://localhost:5000"
LOG_FILE="/home/nickspi5/Chatty_AI/logs/startup.log"
USER="nickspi5"

# Create log directory
mkdir -p /home/nickspi5/Chatty_AI/logs

# Function to log messages
log_message() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# Start logging
log_message "========================================="
log_message "Chatty AI Startup Sequence Initiated"
log_message "========================================="

# Set display
export DISPLAY=:0
export XAUTHORITY=/home/nickspi5/.Xauthority

# Wait for desktop
log_message "Waiting for desktop..."
while ! pgrep -x "pcmanfm" > /dev/null; do
    sleep 1
done
sleep 3

# Kill any error dialogs
pkill -f "Desktop manager" 2>/dev/null
wmctrl -F -c "Desktop manager is not active" 2>/dev/null || true
xdotool search --name "Desktop manager" windowclose 2>/dev/null || true

# Display black screen IMMEDIATELY
log_message "Displaying black screen..."
chromium-browser \
    --kiosk \
    --noerrdialogs \
    --disable-infobars \
    --start-fullscreen \
    --disable-session-crashed-bubble \
    --check-for-update-interval=31536000 \
    "file://$BLACK_PAGE" 2>/dev/null &

BLACK_PID=$!
sleep 2  # Let black screen fully load

# Play video
if [ -f "$VIDEO_PATH" ]; then
    log_message "Playing startup video..."
    
    # Kill black screen BEFORE playing video
    kill $BLACK_PID 2>/dev/null
    pkill -f "black.html" 2>/dev/null
    sleep 0.5
    
    # Play video
    cvlc --fullscreen --play-and-exit "$VIDEO_PATH" 2>/dev/null
    log_message "Video complete"
else
    log_message "Video not found"
    kill $BLACK_PID 2>/dev/null
fi

# Launch loading screen
log_message "Launching loading screen..."

chromium-browser \
    --kiosk \
    --noerrdialogs \
    --disable-infobars \
    --start-fullscreen \
    --check-for-update-interval=31536000 \
    "file://$LOADING_PAGE" 2>/dev/null &

CHROMIUM_PID=$!
log_message "Loading screen PID: $CHROMIUM_PID"

log_message "========================================="

# Keep running
wait $CHROMIUM_PID
