#!/bin/bash

# Hide desktop icons
pcmanfm --desktop-off

chmod +x /home/nickspi5/Chatty_AI/hide_desktop.sh

Then call it at the start of your startup script:

bash# After waiting for desktop

/home/nickspi5/Chatty_AI/hide_desktop.sh

To restore desktop when you press Alt+F4, create:

nano /home/nickspi5/Chatty_AI/show_desktop.sh

#!/bin/bash
# Restore panels
lxpanel --profile LXDE-pi &
# Restore desktop
pcmanfm --desktop --profile LXDE-pi &
